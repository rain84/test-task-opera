import { Canvas, Meta, Story } from '@storybook/addon-docs'
import * as Icons from '../'
const camelCase = /([A-Z]?[a-z-]+)/g
const prepareUrl = (name) =>
  `/?path=/story/icons--${name
    .replace('Icon', '')
    .match(camelCase)
    .join('-')
    .toLocaleLowerCase()}`
const icon = (name, i) => {
  const Icon = Icons[name]
  const onClick = () => ''
  return (
    <div className="p-2 flex space-x-4 hover:bg-elevation hover:bg-opacity-10 rounded-lg">
      <div className="flex-0">{i + 1}.</div>
      <div className="flex-1">
        <a href={prepareUrl(name)}>
          <pre onClick={onClick}>{`<${name} />`}</pre>
        </a>
      </div>
      <div className="flex flex-1">
        <Icon className="mr-2" />
        <Icon className="icon-2x cursor-pointer hover:bg-selected mr-2" />
        <Icon className="icon-white bg-red mr-2" />
      </div>
    </div>
  )
}

<Meta title="UI/Icons" />

# UI / All icons

<Canvas withToolbar>
  <Story name="All-in-One">{Object.keys(Icons).map(icon)}</Story>
</Canvas>
